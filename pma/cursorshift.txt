code segment
assume cs:code,es:code,ds:code,SS:CODE
org 1000h

IR_WR EQU 00H
ST_RD EQU 02H
DR_WR EQU 04H

CALL ALLCLR
y:
CALL CURSOR1
CALL CURSOR2
MOV SI,OFFSET AB
CALL STRING

x:
CALL DELAY

AB DB 'MIST',00H

ALLCLR:
MOV AH,00000001B
JMP OUT

CURSOR1:
MOV AH,080H


OUT:
CALL BUSY
MOV AL,AH
OUT 00H,AL
RET

CURSOR2:
MOV AH,00011100B
CALL BUSY
MOV AL,AH
OUT 00H,AL
RET


BUSY:
MOV DX,ST_RD
IN AL,DX
TEST AL,10000000B
JNZ BUSY
RET

STRING:
MOV AH,CS:[SI]
CMP AH,00H
JE RETURN

CALL BUSY

MOV AL,AH
MOV DX,DR_WR
OUT DX,AL
INC SI
JMP STRING
RETURN :
RET

DELAY:
MOV CX,0FFFFH
F:DEC CX
JNZ F

JMP y

CODE ENDS
END
